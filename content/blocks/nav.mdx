import { useNav } from "@reflexjs/gatsby-theme-nav"

export const Nav = () => {
  const [main] = useNav('main')
  return main.items ? (
    <Ul>
      {main.items.map(({value, url, items}, index) => (
        <Li key={index}>
          <Link href={url}>{value}</Link>
        </Li>
      ))}
      </Ul>
  ) : null
}
